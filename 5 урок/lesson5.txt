SELECT

from users;

2.


SELECT
DATE_FORMAT(DATE(CONCAT_WS('-', YEAR(NOW()), MONTH(birthday_at), DAY(birthday_at))), '%W') AS day,
COUNT(*) AS total
from
 users
 GROUP BY
 day
 ORDER BY 
	total DESC;
